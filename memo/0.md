# 合否判定をメソッドチェーンで表記できるAPIを作りたい

`String.prototype.indexOf()`などでは成功時と失敗時に分かれる。しかし失敗時には例外発生せず、成功時と同じNumber型プリミティブ値`-1`を返す。この時、成否を`-1!==res`のようにマジックナンバーで判定することになる。これは可読性が低いため回避したい。他にも成功時はインスタンスを返し、失敗時にはnullを返すようなAPIもある。いずれにせよ成功／失敗を具体的な値ではなく抽象的な値を間に噛ませて一般化できるAPIを作りたい。

Promiseの真似をする。あるいはRustのmatch式の二択版みたいなもの。

```javascript
// 文字列の中から指定した字のインデックスを取得する。
// 実行結果を成功か失敗かで大別できる。
const indexOf = (s, c) => {
	return new Judgement((success, fail)=>{
		const i = s.indexOf(c);
		return 0<=i ? success(i) ? fail(new Error(`指定の字'${c}'は存在しませんでした。`))
	});
};

// Promise方式
indexOf('abcdef', 'd')
.success(res=>res)
.fail(e=>throw e);

// match方式
const index = indexOf('abcdef', 'd').judge(res=>res, e=>-1);
```
```javascript
class Judgement {
	static success(v) {return new Success(v)}
	static fail(v) {return new Fail(v)}
	constructor(fn) {
		if ('function'!==typeof fn) {throw new TypeError(`fnは関数であるべきです。`)}
		const res = fn('success fail'.split(' ').map(n=>Judgement[n]));
		if (![Success,Fail].some(C=>res instanceof C)) {throw new TypeError(`fnの戻り値はSuccessかFailのいずれかであるべきです。`)}
		this._={fn:fn, res:res};
	}
	get isSuccess() {return this._.res instanceof Success}
	get isFail() {return this._.res instanceof Fail}
	judge(success, fail) {
		const fn = this.isSuccess ? success : fail;
		const arg = this._.res[this.isSuccess ? 'value' : 'cause'];
		return fn(arg);
	}
	success(fn) {
		if (this._.res instanceof Success) {
			fn(this._.res.value);
		}
		return this;
	}
	fail(fn) {
		if (this._.res instanceof Fail) {
			fn(this._.res.cause);
		}
		return this;
	}
}
class JudgementResult {}
class Success extends JudgementResult {
	constructor(value) {this._={value}}
	get value() {return this._.value}
}
class Fail extends JudgementResult {
	constructor(cause) {this._={cause}}
	get cause() {return this._.cause}
	throw(error) {throw (error instanceof Error ? error : this._.cause)}
}
```

`success`は長いので`pass`に改名するか。

```javascript
// 文字列の中から指定した字のインデックスを取得する。
// 実行結果を成功か失敗かで大別できる。
const indexOf = (s, c) => {
	return new Judgement((pass, fail)=>{
		const i = s.indexOf(c);
		0<=i ? pass(i) ? fail(new Error(`指定の字'${c}'は存在しませんでした。`))
	});
};

// Promise方式
indexOf('abcdef', 'd')
.pass(v=>v)
.fail(e=>throw e);
.any(j=>j.isPass ? (()=>{console.log('成功')}) : (()=>{console.log('失敗')}););

// match方式
const index = indexOf('abcdef', 'd').judge(v=>v, e=>-1);
```
```javascript
class Judgement {
	static pass(v) {return new Pass(v)}
	static fail(v) {return new Fail(v)}
	static isFn(v, n) {return 'function'==typeof v}
	static throwFn(v, n) {if (!this.isFn(v)) {throw new TypeError(`${n}は関数であるべきです。`)}}
	constructor(fn) {
		Judgement.throwFn(fn, 'fn');
		const res = fn('pass fail'.split(' ').map(n=>Judgement[n]));
		if (![Pass,Fail].some(C=>res instanceof C)) {throw new TypeError(`fnの戻り値はPassかFailのいずれかであるべきです。`)}
		this._={fn:fn, res:res};
	}
	get isPass() {return this._.res instanceof Pass}
	get isFail() {return this._.res instanceof Fail}
	get result() {return this._.res}
	judge(pass, fail) {
		[pass, fail].forEach((fn,i)=>Judgement.throwFn(fn, 0===i ? 'pass' : 'fail'));
		const fn = this.isPass ? pass : fail;
		const arg = this._.res[this.isPass ? 'value' : 'cause'];
		return fn(arg);
	}
	pass(fn) {return #some('Pass', fn);}
	fail(fn) {return #some('Fail', fn);}
	#some(name, fn) {
		Judgement.throwFn(fn, 'fn');
		if (this[`is${name}`]) {fn(this._.res['Pass'===name ? 'value' : 'cause']);}
		return this;
	}
	any(fn) {fn(this._.res); return this;}
}
class JudgementResult {}
class Pass extends JudgementResult {
	constructor(value) {this._={value}}
	get value() {return this._.value}
}
class Fail extends JudgementResult {
	constructor(cause) {this._={cause}}
	get cause() {return this._.cause}
	throw(error) {throw (error instanceof Error ? error : this._.cause)}
}
```
